project("assets")

include(${CMAKE_CURRENT_SOURCE_DIR}/scripts/bin2h.cmake)

add_library(${PROJECT_NAME} INTERFACE)

file(GLOB_RECURSE IMAGE_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/*.png
)

foreach(imageFile ${IMAGE_SOURCES})
    get_filename_component(imageName ${imageFile} NAME_WE)
    bin2h(SOURCE_FILE ${imageFile} 
          HEADER_FILE ${CMAKE_CURRENT_BINARY_DIR}/assets_${imageName}.h 
          VARIABLE_NAME "assets_${imageName}_png" 
          NULL_TERMINATE)
    set_property(DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}" APPEND PROPERTY CMAKE_CONFIGURE_DEPENDS 
        ${imageFile}
    )
    message(STATUS "Embedded ${imageFile} as assets_${imageName}_png in ${CMAKE_CURRENT_BINARY_DIR}/assets_${imageName}.h")
endforeach()

set_property(DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}" APPEND PROPERTY CMAKE_CONFIGURE_DEPENDS 
    ${CMAKE_CURRENT_SOURCE_DIR}
)

set(INCLUDES
    ${CMAKE_CURRENT_BINARY_DIR}
)

target_include_directories(${PROJECT_NAME} INTERFACE ${INCLUDES})
